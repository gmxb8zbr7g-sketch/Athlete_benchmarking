                }

        function calculateCriteriaPercentile(val, m) {
            const { concern, standard, elite } = m;
            if (m.lowerIsBetter) {
                if (val >= concern) return 5;
                if (val <= elite) return 95;
                if (val > standard) return Math.round(10 + (1 - (val - standard) / (concern - standard)) * 40);
                return Math.round(50 + (1 - (val - elite) / (standard - elite)) * 40);
            } else {
                if (val <= concern) return 5;
                if (val >= elite) return 95;
                if (val < standard) return Math.round(10 + ((val - concern) / (standard - concern)) * 40);
                return Math.round(50 + ((val - standard) / (elite - standard)) * 40);
            }
        }

        const Icon = ({ name, size = 18, className = "" }) => {
            const [svg, setSvg] = useState("");
            useEffect(() => {
                if (window.lucide) {
                    const iconNode = window.lucide.icons[name];
                    if (iconNode) {
                        const svgElement = document.createElementNS("http://www.w3.org/2000/svg", "svg");
                        Object.entries(iconNode[1]).forEach(([k, v]) => svgElement.setAttribute(k, v));
                        iconNode[2].forEach(child => {
                            const childElement = document.createElementNS("http://www.w3.org/2000/svg", child[0]);
                            Object.entries(child[1]).forEach(([k, v]) => childElement.setAttribute(k, v));
                            svgElement.appendChild(childElement);
                        });
                        svgElement.setAttribute("width", size);
                        svgElement.setAttribute("height", size);
                        svgElement.setAttribute("class", className);
                        setSvg(svgElement.outerHTML);
                    }
                }
            }, [name, size, className]);
            return <span dangerouslySetInnerHTML={{ __html: svg }} />;
        };

                    <p className="text-sm font-bold text-slate-800 truncate">{value}</p>
                </div>
            </div>

        const App = () => {

            const results = useMemo(() => {

                    const percentile = calculateCriteriaPercentile(val, m);
                    
                    let status, color;
                    if (percentile >= 90) { status = 'Elite'; color = '#a855f7'; } 
                    else if (percentile >= 75) { status = 'Target'; color = '#22c55e'; } 
                    else if (percentile >= 40) { status = 'Standard'; color = '#3b82f6'; } 
                    else if (percentile >= 15) { status = 'Below'; color = '#eab308'; } 
                    else { status = 'Concern'; color = '#ef4444'; }

                    return { key, label: m.name, value: val, standardValue: m.standard, percentile, status, color };
                });
            }, [athleteData, cohortData]);

            const radarData = results.map(r => ({
                metric: r.label, Value: r.percentile, Elite: 90, Standard: 50, Concern: 15
            }));

            const generateRandomData = () => {
                const newData = {};
                Object.keys(cohortData.metrics).forEach(key => {

                    const variance = 0.8 + Math.random() * 0.4;

                    }}}

                            <div>
                                <h1 className="text-3xl font-bold flex items-center gap-2 tracking-tight uppercase text-slate-800">

                                        key={tab}

                                        const m = norms.custom.male.metrics[key];

                                            </div>
                                        );
                                    })}
                                </div>
                                <div className="mt-8 flex justify-end"><button onClick={() => { setProfile({ sport: 'custom', sex: 'male' }); setActiveTab('input'); }} className="px-8 py-4 bg-indigo-600 text-white rounded-xl font-bold uppercase text-xs shadow-lg hover:bg-indigo-700 transition-all flex items-center gap-2">Apply Settings <Icon name="Zap" size={16} /></button></div>
                            </div>
                        )}
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
